example​

// setup Component
defineComponent({
  setup(props, { attrs, slots, emit, expose }) {
    ...
    return {}   // 合并data
    // return ()=>{}   render
  },
})

<!-- script setup   -->

<script setup>
    const props = defineProps({
        title: String
    })
    // props.title
    const { slots, attrs } = useContext()
    // 获取 emit
    const emit = defineEmit(["chang-name"])
    // 调用 emit
    emit("chang-name", "Tom")
</script>
